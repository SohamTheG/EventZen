// --- USER & ATTENDEE SERVICE ---

Table users {
  id bigint [primary key, increment]
  email varchar [not null]
  password varchar
  name varchar
  role varchar
}

Table attendees {
  id bigint [primary key, increment]
  user_id bigint
  event_id int [not null]
  status varchar
}

// --- EVENTS & BOOKING SERVICE ---

Table events {
  id bigint [primary key, increment]
  name varchar
  description varchar
  host_id bigint
  is_public bit [not null]
}

Table bookings {
  id bigint [primary key, increment]
  event_date date
  status enum // 'APPROVED', 'CANCELLED', 'PENDING', 'REJECTED'
  venue_id bigint
  event_id bigint [unique]
}

// --- VENUE SERVICE ---

Table venues {
  id bigint [primary key, increment]
  name varchar [not null]
  location varchar [not null]
  capacity int
  price_per_day decimal
  createdAt datetime
  updatedAt datetime
}

Table vendors {
  id bigint [primary key, increment]
  name varchar [not null]
  type varchar [not null]
  createdAt datetime
  updatedAt datetime
}

Table venuevendors {
  VenueId bigint [not null]
  VendorId bigint [not null]
  createdAt datetime
  updatedAt datetime
  
  indexes {
    (VenueId, VendorId) [pk]
  }
}
